62  constexpr uint8_t hexCharToInt(char c) {
63  if (c >= '0' && c <= '9')
64  return gsl::narrow<uint8_t>(c - '0');
65  else if (c >= 'a' && c <= 'f')
66  return gsl::narrow<uint8_t>(10 + c - 'a');
67  else if (c >= 'A' && c <= 'F')
68  return gsl::narrow<uint8_t>(10 + c - 'A');
